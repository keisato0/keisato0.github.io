for (i in 1:length(elections_year_list)) {
    tentative <- semi_join(elections_year_list[[i]], map_county_list, 
        by = "state_county")
    png(filename = paste(as.character(1956 + i * 4), "png", 
        sep = "."), width = 800, height = 600)
    map(database = "county")
    for (j in 1:nrow(tentative)) {
        map(database = "county", regions = tentative$state_county[j], 
            col = rgb(red = tentative$rep_ratio[j], blue = tentative$dem_ratio[j], 
              green = 0), add = TRUE, fill = TRUE)
    }
    dev.off()
}
